SELECT jobs_files.ID, jobs_files.Filename, jobs_files.Pages, CONCAT(jobs_files.Deadline,' ',jobs_files.DeadlineTime) as Deadline, jobs_files.AltTxt, jobs_files.JobID, jobs.JobType,  (Select CONCAT(YEAR(ClientCreated),'-',Country,'-',Code,'-',jobs.ID) from clients  RIGHT JOIN jobs ON jobs.ClientID = clients.ID Where jobs.ID = JOBID) EngagementNum  FROM jobs_files  LEFT JOIN jobs ON jobs.ID = jobs_files.JobID LEFT JOIN clients ON clients.ID = jobs.ClientID WHERE 1 AND jobs.Status = 'OPEN' AND jobs_files.Status <> 'CLOSED' AND jobs_files.Status <> 'PENDING' AND(jobs_files.AltTxt IS NULL OR jobs_files.AltTxt = 0 OR jobs_files.AltTxt = 2 OR jobs_files.AltTxt = 6) AND jobs_files.FolderFlagged <> 2 AND jobs_files.ALTCheckout = 0 AND jobs_files.Deleted = 0 AND (jobs_files.ID Like '%6239%' OR jobs_files.JobID Like '%6239%' OR jobs_files.Filename Like '%6239%' OR CONCAT(YEAR(ClientCreated), '-', Country, '-', Code, '-', jobs.ID) like '%6239%') ORDER BY jobs_files.Deadline, jobs_files.DeadlineTime, FIELD(jobs.JobType, 'ASAP', 'RUSH', 'FIXED', 'FLEX', 'MULTI') LIMIT @end OFFSET @start
